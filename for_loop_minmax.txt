#!/bin/bash

numbers=()
for i in {1..3}; do
    while true; do
        echo "Enter number $i: "
        read -r num
        if [[ "$num" =~ ^-?[0-9]+$ ]]; then
            numbers+=("$num")
            break
        else
            echo "Error: Please enter a valid integer."
        fi
    done
done

max=${numbers[0]}
min=${numbers[0]}
for num in "${numbers[@]}"; do
    if [ "$num" -gt "$max" ]; then max=$num; fi
    if [ "$num" -lt "$min" ]; then min=$num; fi
done

echo "max = $max"
echo "min = $min"
