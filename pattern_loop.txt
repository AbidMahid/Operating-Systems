#!/bin/bash

echo "Enter a number (1-6): "
read num

# Check if the number is between 1 and 6
if [ "$num" -lt 1 ] || [ "$num" -gt 6 ]; then
    echo "Error: Please enter a number between 1 and 6."
    exit 1
fi

# Loop to print the pattern
for ((i = 1; i <= num; i++)); do
    line=""
    # Add the number i, i times
    for ((j = 1; j <= i; j++)); do
        line="$line$i "
    done
    # Remove extra space and print
    echo "${line% }"
done
